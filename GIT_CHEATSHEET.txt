┌─────────────────────────────────────────────────────────┐
│ 🚀 COMANDI PIÙ USATI                                    │
└─────────────────────────────────────────────────────────┘

📝 COMMIT RAPIDO (feature branch)
   ./quick-commit.sh                    # Modalità interattiva
   ./quick-commit.sh "feat: descrizione" # Con messaggio

🌐 DEPLOY ONLINE (GitHub Pages)
   ./deploy.sh

🧪 TEST LOCALE
   ./test-local.sh

📊 VERIFICA STATO
   ./check-deploy.sh

🔢 AGGIORNA VERSIONE
   ./bump-version.sh patch    # 0.1.0 → 0.1.1
   ./bump-version.sh minor    # 0.1.0 → 0.2.0
   ./bump-version.sh major    # 0.1.0 → 1.0.0


┌─────────────────────────────────────────────────────────┐
│ 📋 COMANDI MANUALI                                      │
└─────────────────────────────────────────────────────────┘

➕ AGGIUNGERE MODIFICHE
   git add .

💾 COMMIT
   git commit -m "feat: descrizione"

🚀 PUSH
   git push

🔄 SWITCH BRANCH
   git checkout main
   git checkout feature/revolution-game

🔀 MERGE & DEPLOY
   git checkout main
   git merge feature/revolution-game
   git push origin main


┌─────────────────────────────────────────────────────────┐
│ 📊 VERIFICA STATO                                       │
└─────────────────────────────────────────────────────────┘

git status              # Vedi file modificati
git log --oneline -5    # Ultimi 5 commit
git branch              # Lista branch


┌─────────────────────────────────────────────────────────┐
│ 🔙 ANNULLA OPERAZIONI                                   │
└─────────────────────────────────────────────────────────┘

git restore file.js          # Annulla modifiche file
git reset --soft HEAD~1      # Annulla ultimo commit
git stash                    # Salva modifiche temporanee
git stash pop                # Recupera modifiche salvate
